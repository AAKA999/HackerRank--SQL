--MS SQL SERVER
SELECT A.HACKER_ID, NAME, SUM(A.SCORE)
FROM (
    SELECT S.HACKER_ID, NAME, CHALLENGE_ID, MAX(SCORE) SCORE
    FROM HACKERS H
    JOIN SUBMISSIONS S
    ON H.HACKER_ID=S.HACKER_ID
    GROUP BY CHALLENGE_ID, S.HACKER_ID, NAME
    HAVING MAX(SCORE)>0
    ) A
GROUP BY HACKER_ID, NAME
ORDER BY SUM(A.SCORE) DESC, HACKER_ID
